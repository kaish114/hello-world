#include <stdio.h>

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char a[10],b[10];
        int sum[10];
        scanf("%s%s",&a,&b);
        int k=strlen(a)>strlen(b)?strlen(a):strlen(b);
        int g=k;
        k--;
        for(int i=strlen(a)-1,j=strlen(b)-1;k>=0;i--,j--,k--)
        {
            sum[k]=(int)(a[i]+b[i]-6)%10;
        }
        for(int i=0;i<g;i++)
        {
            if(sum[i]>0)
        printf("%d",sum[i]);
        else
        printf("%d",sum[i]+10);
        }
        printf("\n");
    }
    return 0;
}
